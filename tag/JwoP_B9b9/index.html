public Boolean getAssessResult(String componentName, String bridgeId,String projectNo,Integer partId) {
        //判断桥梁是否已经检测完毕了
        LqBridge lqBridge = lqBridgeDao.selectById(bridgeId);
        if(lqBridge.getIsCheck()==0){
            return false;
        }
        //这里就是检测完毕的，可以进行评测,去了重，算了平均值，已有diseaseName,diseaseId，scale,

        List<Map<String,Object>> assessList = lqDiseaseAssessDefineService.getAssessByCompenentName(componentName,bridgeId,projectNo,partId);

        List<LqDiseaseAssessDto> assessDtoList = new ArrayList<>();
        // 到按照给定的标度和病害名称来添加标度问题
        for (Map<String, Object> assess : assessList) {
            LqDiseaseAssessDto dto = new LqDiseaseAssessDto();
            dto.setDiseaseID((String)assess.get("diseaseId"));
            String avgScale = String.valueOf(assess.get("avgScale"));
            avgScale=avgScale.substring(0,avgScale.indexOf("."));
            dto.setScale(Integer.parseInt(avgScale));
            assessDtoList.add(dto);
        }

        //获取扣分值
        Map<String, String> deductByScale = lqDiseaseAssessDefineService.getDeductByScale(assessDtoList);

        //拿到扣分值，然后根据diseaseId进行赋值
        for (Map<String, Object> stringObjectMap : assessList) {
                Iterator<Map.Entry<String, String>> iterator = deductByScale.entrySet().iterator();
                while (iterator.hasNext()){
                    Map.Entry<String, String> dedule = iterator.next();
                    String diseaseId = dedule.getKey();
                    String deductPoint = dedule.getValue();

                    if(stringObjectMap.get("diseaseId").equals(diseaseId)){
                        stringObjectMap.put("deductPoint",deductPoint);
                        break;
                    }
                }

        }
        //获取这个病害的数量
        Integer number = lqDiseaseAssessDefineService.diseaseCountByComponent(componentName,partId);


        //pmci计算值
        Double pmci = PMCI(deductByScale,number);



        //验证是否有值
        for (Map<String, Object> user : assessList) {
            System.out.println(user.get("diseaseId") + " " + user.get("diseaseName") + " " + user.get("deductPoint"));

        }
        System.out.println("pmci = " + String.format("%.1f",(100-pmci)));


        //剩下就是保存了。

        return true;
    }
