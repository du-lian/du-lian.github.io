        //剩下就是保存了。

        return true;

public static List<LqDiseaseAssessDto> addScaleByScaleDiseaseName(List<Map<String, Object>> assessList) {

 public static List<Map<String, Object>> deductByDiseaseId(List<Map<String, Object>> assessList,Map<String, String> deductByScale) {
        for (Map<String, Object> stringObjectMap : assessList) {
            Iterator<Map.Entry<String, String>> iterator = deductByScale.entrySet().iterator();
            while (iterator.hasNext()){
                Map.Entry<String, String> dedule = iterator.next();
                String diseaseId = dedule.getKey();
                String deductPoint = dedule.getValue();

                if(stringObjectMap.get("diseaseId").equals(diseaseId)){
                    stringObjectMap.put("deductPoint",deductPoint);
                    break;
                }
            }
        }
        return assessList;
    }

public static Object addField(Object obj,Class<?> clazz,List<Map<String,Object>> assessList,String tableName) {
        List tableNames=new ArrayList();
        tableNames.add(tableName);
        Map<String, String> columnCommentByTableName = null;
        try {
            //这个是不转的，因为只有知道这个才知道 注释和 字段名一致
            columnCommentByTableName = MySqlUtils.getColumnCommentByTableName(tableNames);

            for (Map.Entry<String, String> entry : columnCommentByTableName.entrySet()) {
                String key = entry.getKey();
                if(key.contains("评定")){
                    //评定的
                    String assessField = entry.getValue();
                    //这里要判断那个Field
                    String subKey = key.replace("评定", "");
                    for (Map<String, Object> stringObjectMap : assessList) {
                        if(stringObjectMap.get("diseaseName").equals(subKey)){
                            try {
                                //这里有个问题，如果大写了，那个字段就找不到了。
                                String AssessField = BridgeUtils.stringToUpperCase(assessField);
                                int avgScale = Integer.parseInt(String.valueOf(stringObjectMap.get("avgScale")));
                                ReflectUtils.setValue(obj, clazz, AssessField, clazz.getDeclaredField(assessField).getType(),avgScale );
                                //顺便把扣分也做了
                                String deductedField = assessField.replace("assess", "deducted");
                                String DeductedField = BridgeUtils.stringToUpperCase(deductedField);
                                int deductPoint = Integer.parseInt(String.valueOf(stringObjectMap.get("deductPoint")));
                                ReflectUtils.setValue(obj, clazz, DeductedField, clazz.getDeclaredField(deductedField).getType(), deductPoint);
                            } catch (NoSuchFieldException e) {
                                throw new RuntimeException(e);
                            }
                        }
                    }
                }
            }
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
        return obj;
    }
