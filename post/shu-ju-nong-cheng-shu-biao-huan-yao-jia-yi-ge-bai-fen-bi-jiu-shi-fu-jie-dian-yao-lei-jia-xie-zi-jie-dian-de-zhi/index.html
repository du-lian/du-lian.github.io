数据弄成树表，还要加一个百分比，就是父节点要累加叶子节点的值

2022-09-21 工作遇到的难点

树表的构成，我一开始想的是递归的， 但后面想了一下，好像不太好实现，特别我的那个百分比的累加怎么搞？

基于我的数据库的结构

我的数据库是有isLeaf 和 parentID的字段。

可以使用stream流

{
    //1.一开始先找到所需要的信息
    xxx.listVo=dao.findAllInfo(xxx);
    //2.这样就可以开始stream流的开始
    listVo.parallelStream()
        .filter(item-> item.getParentID().equals("-1"))//总结点ID为0
                    //这一步是item里面有一个子的List的， 然后 
                    //item就是最上面的结点， list就是全部数据
                    //这样， 就可以直接找item下面的孩子结点
        .peek((item)->{item.setxxxVOList(getChildren(item,list));})
        .cooect(Collectors.toList());
}
/**
*@param root     上面的总结点
* @param all     全部列表
*/
List<xxxVO> getChildren(xxx root, List<xxx> all){
    return all.stream()
        .filter(item-> item.getParentID.equals(root.getID))//用item查找是否属于总结点的孩子结点
        .peek(item->{
            //从这里迭代
            item.xxxVolist(getChildren(item,all))
        })
        .collect(Collectors.toList());
}

这样就可以得到一个树型结构了。但我还要再加个百分比。

就是父节点要累加叶子节点的值

这能在这个基础上做修改了

{
    //1.一开始先找到所需要的信息
    xxx.listVo=dao.findAllInfo(xxx);
    //1.1因为我要和某个数据做对比，我又不想申请多次数据库请求，我就一开始直接拿那个数据的全部数据出来，存入map，用要对比的字段作为key， 某一个值作为value。
    Map<String,String> allCxx = xxx.findAllCxxxList();
    
    //2.这样就可以开始stream流的开始
    listVo.parallelStream()
        .filter(item-> item.getParentID().equals("-1"))//总结点ID为0
                    //这一步是item里面有一个子的List的， 然后 
                    //item就是最上面的结点， list就是全部数据
                    //这样， 就可以直接找item下面的孩子结点
        .peek((item)->{
            //重写了一个getChildren功能
            item.setxxxVOList(getChildren(item,list,allCxx));
            //这里要添加一些判断语句，每个人不一样，
            if(item.getxxxList()!=null){
                int len = item.getxxxList().size();
                        int num = 0; //用于判断下面的item是否等于100？1：0

                        //累和子节点的百分比，放置父节点的百分比
                        for (xxxVO vo :item.getxxxVOList()){
                            if (vo.getPercentum().equals(Double.valueOf(100))) {
                                num++;
                            }
                        }
                        item.setPercentum(Math.floor(num/(len*1.0)*100));
            }else{
                //因为如果子孩子没有的话，可以直接判断为空
                item.setPercentum(Double.valueOf(0));
            }
        })
        .collect(Collectors.toList());
}
/**
*@param root     上面的总结点
* @param all     全部列表
*/
List<xxxVO> getChildren(xxx root, List<xxx> all,Map<String,String> allxxList){
    return all.stream()
        .filter(item-> item.getParentID.equals(root.getID))//用item查找是否属于总结点的孩子结点
        .peek(item->{
            //从这里迭代
            item.xxxVolist(getChildren(item,all,allxxList))
            //添加一个百分比的功能
            item = addPercentumLeaf(item,allxxxList);
                
        })
        .collect(Collectors.toList());
}

/*
     * @description: 给list的叶子结点添加一个计算百分比功能
     */
    private xxxVO addPercentumLeaf(xxx item, Map<String,String> allxxxList) {
        //是 root的孩子结点就可以进来了。
        if("1".equals(item.getIsLeaf())){ //叶子结点
            if(item.getProjectID()!=null && item.getId()!=null){
//                    这里就是叶子结点了
//                    开始判断 物业清单，协作清单，协作清单
                if(allxxxList.containsKey(item.getId())){
                    item.setPercentum(Double.valueOf(100));
                }else{
                    item.setPercentum(Double.valueOf(0));
                }
            }else {
                item.setPercentum(Double.valueOf(0));
            }
        }else{
            //非叶子结点
            if(item.getManageUnitVOList()!=null){
                int len = item.getxxxList().size();
                int num = 0; //用于判断下面的item是否等于100？1：0

                //累和子节点的百分比，放置父节点的百分比
                for (xxxVO vo :item.getxxxList()){
                    if (vo.getPercentum().equals(Double.valueOf(100))) {
                        num++;
                    }
                }
                item.setPercentum(Math.floor(num/(len*1.0)*100));

            }else{
                item.setPercentum(Double.valueOf(0));
            }
        }

        item.setPercentum(item.getPercentum()==null? 0:item.getPercentum());

        return item;
    }
